    function renderCharts(rows){
      // Day trend
      const byDay = countBy(rows, r => r.date)
      drawBar("chByDay", byDay.labels.map(fmtDate), byDay.values)

      // Hour bars

    function drawBar(canvasId, labels, values){
      const c = document.getElementById(canvasId)
      const ctx = c.getContext("2d")
      const w = c.width
      const h = c.height

      ctx.clearRect(0,0,w,h)

      const padL = 36
      const padR = 10
      const padT = 10
      const padB = 26

      const maxV = Math.max(1, ...values)
      const n = Math.max(1, values.length)

      // Axes
      ctx.strokeStyle = "#cbd5e1"
      ctx.lineWidth = 1
      ctx.beginPath()
      ctx.moveTo(padL, padT)
      ctx.lineTo(padL, h - padB)
      ctx.lineTo(w - padR, h - padB)
      ctx.stroke()

      const plotW = (w - padL - padR)
      const plotH = (h - padT - padB)
      const gap = 6
      const barW = Math.max(2, (plotW - gap*(n-1)) / n)

      // Bars
      for (let i=0;i<n;i++){
        const v = values[i] || 0
        const x = padL + i*(barW + gap)
        const bh = Math.round((v / maxV) * plotH)
        const y = (h - padB) - bh

        ctx.fillStyle = "#0d233c"
        ctx.fillRect(x, y, barW, bh)

        // Label tick
        if (n <= 12){
          ctx.fillStyle = "#5b6678"
          ctx.font = "10px " + getComputedStyle(document.documentElement).getPropertyValue("--sans")
          const t = (labels[i] || "").slice(0, 10)
          ctx.fillText(t, x, h - 10)
        }
      }

      // Max label
      ctx.fillStyle = "#5b6678"
      ctx.font = "10px " + getComputedStyle(document.documentElement).getPropertyValue("--mono")
      ctx.fillText(String(maxV), 6, 14)
    }

    function countBy(rows, keyFn){
      const m = new Map()
      rows.forEach(r => {
        const k = keyFn(r) || "Unknown"
        m.set(k, (m.get(k) || 0) + 1)
      })

      const labels = Array.from(m.keys()).sort()
      const values = labels.map(k => m.get(k))
      return { labels, values }
    }

    function topN(labels, values, n){
      const pairs = labels.map((l,i) => ({l, v: values[i] || 0}))
      pairs.sort((a,b) => b.v - a.v)
      const cut = pairs.slice(0, n)
      return { labels: cut.map(x => x.l), values: cut.map(x => x.v) }
    }

      const out = rows.slice()
      out.sort((a,b) => {
        const av = (a[key] ?? "").toString()
        const bv = (b[key] ?? "").toString()
        if (av < bv) return (dir === "asc") ? -1 : 1
        if (av > bv) return (dir === "asc") ? 1 : -1
        return 0
      })
      return out
    }

    function unique(arr){
      return Array.from(new Set(arr)).sort()
    }

    function fmtDate(iso){
      if (!iso) return ""
      const parts = iso.split("-")
      if (parts.length !== 3) return iso
      return parts[1] + "/" + parts[2] + "/" + parts[0]
    }

    function scopeLabel(){
      const o = document.getElementById("fOfficer").value
      const t = document.getElementById("fType").value
      const pg = document.getElementById("fPostGrid").value
      const bits = []
      if (o !== "All") bits.push("Officer " + o)
      if (t !== "All") bits.push("Type " + t)
      if (pg !== "All") bits.push(pg)
      return bits.length ? bits.join(" | ") : "All"
    }

    function exportCsv(rows){
      const cols = ["case_id","date","time","type","officer","post_grid","location","person_id","risk_flag","duration_min","lat","lon","narrative"]
      const lines = []
      lines.push(cols.join(","))

      rows.forEach(r => {
        const line = cols.map(k => csvCell(r[k])).join(",")
        lines.push(line)
      })

      const blob = new Blob([lines.join("\n")], { type:"text/csv;charset=utf-8" })
      const url = URL.createObjectURL(blob)

      const a = document.createElement("a")
      a.href = url
      a.download = "FCE_PA_Report.csv"
      document.body.appendChild(a)
      a.click()
      a.remove()
      URL.revokeObjectURL(url)
    }

    function csvCell(v){
      const s = (v === null || v === undefined) ? "" : String(v)
      const q = s.replace(/"/g,'""')
      return '"' + q + '"'
    }
